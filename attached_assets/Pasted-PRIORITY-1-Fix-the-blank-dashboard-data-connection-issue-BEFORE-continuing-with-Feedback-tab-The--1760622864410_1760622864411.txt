```typescript
import { createClient } from '@/lib/supabase/server'
import { NextResponse } from 'next/server'

export async function GET() {
  try {
    const supabase = createClient()
    
    // Get restaurant count
    const { count: restaurantCount } = await supabase
      .from('restaurants')
      .select('*', { count: 'exact', head: true })
      .eq('status', 'active')
    
    // Get user count
    const { count: userCount } = await supabase
      .from('users')
      .select('*', { count: 'exact', head: true })
    
    // Get orders for revenue and count
    const { data: orders } = await supabase
      .from('orders')
      .select('total_amount, created_at, order_status')
    
    const totalRevenue = orders?.reduce((sum, order) => sum + (Number(order.total_amount) || 0), 0) || 0
    const totalOrders = orders?.length || 0
    
    // Get recent orders with restaurant and user details
    const { data: recentOrders } = await supabase
      .from('orders')
      .select(`
        id,
        uuid,
        total_amount,
        order_status,
        created_at,
        restaurant_id,
        user_id
      `)
      .order('created_at', { ascending: false })
      .limit(10)
    
    // Get restaurant names for recent orders
    const restaurantIds = recentOrders?.map(o => o.restaurant_id) || []
    const { data: restaurants } = await supabase
      .from('restaurants')
      .select('id, name')
      .in('id', restaurantIds)
    
    // Get user emails for recent orders
    const userIds = recentOrders?.map(o => o.user_id) || []
    const { data: users } = await supabase
      .from('users')
      .select('id, email')
      .in('id', userIds)
    
    // Map restaurant and user names to orders
    const ordersWithDetails = recentOrders?.map(order => ({
      ...order,
      restaurant_name: restaurants?.find(r => r.id === order.restaurant_id)?.name || 'Unknown',
      user_email: users?.find(u => u.id === order.user_id)?.email || 'Unknown'
    })) || []
    
    return NextResponse.json({
      success: true,
      stats: {
        totalRevenue: totalRevenue.toFixed(2),
        totalOrders,
        totalRestaurants: restaurantCount || 0,
        totalUsers: userCount || 0
      },
      recentOrders: ordersWithDetails,
      timestamp: new Date().toISOString()
    })
  } catch (error: any) {
    console.error('Dashboard stats error:', error)
    return NextResponse.json({ 
      success: false,
      error: 'Failed to fetch dashboard stats',
      details: error?.message 
    }, { status: 500 })
  }
}
```
